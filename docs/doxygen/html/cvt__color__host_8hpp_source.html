32void rgb_or_bgr_to_yuv(SRC input, DST output, int64_t width, int64_t height,
33 int64_t batch_size, int orderIdx, float delta) {
34 for (int z_idx = 0; z_idx < batch_size; z_idx++) {
35 for (int y_idx = 0; y_idx < height; y_idx++) {
36 for (int x_idx = 0; x_idx < width; x_idx++) {
37 T R = input.template at<T>(z_idx, y_idx, x_idx, orderIdx);
38 T G = input.template at<T>(z_idx, y_idx, x_idx, 1);
39 T B = input.template at<T>(z_idx, y_idx, x_idx, orderIdx ^ 2);
40
41 float Y = R * 0.299f + G * 0.587f + B * 0.114f;
42 float Cr = (R - Y) * 0.877f + delta;
43 float Cb = (B - Y) * 0.492f + delta;
44
45 output.template at<T>(z_idx, y_idx, x_idx, 0) =
46 RoundImplementationsToYUV<float>(Y);
47 output.template at<T>(z_idx, y_idx, x_idx, 1) =
48 RoundImplementationsToYUV<float>(Cb);
49 output.template at<T>(z_idx, y_idx, x_idx, 2) =
50 RoundImplementationsToYUV<float>(Cr);
51 }
52 }
53 }

56template <typename T, typename SRC, typename DST>
57void yuv_to_rgb_or_bgr(SRC input, DST output, int64_t width, int64_t height,
58 int64_t batch_size, int orderIdx, float delta) {
59 for (int z_idx = 0; z_idx < batch_size; z_idx++) {
60 for (int y_idx = 0; y_idx < height; y_idx++) {
61 for (int x_idx = 0; x_idx < width; x_idx++) {
62 T Y = input.template at<T>(z_idx, y_idx, x_idx, 0);
63 T Cb = input.template at<T>(z_idx, y_idx, x_idx, 1);
64 T Cr = input.template at<T>(z_idx, y_idx, x_idx, 2);
65
66 float B = Y + (Cb - delta) * 2.032f;
67 float G = Y + (Cb - delta) * -0.395f + (Cr - delta) * -0.581f;
68 float R = Y + (Cr - delta) * 1.140f;
69
70 output.template at<T>(z_idx, y_idx, x_idx, orderIdx) =
71 Clamp<T, float>(RoundImplementationsFromYUV<float>(R), 0,
72 255);
73 output.template at<T>(z_idx, y_idx, x_idx, 1) = Clamp<T, float>(
74 RoundImplementationsFromYUV<float>(G), 0, 255);
75 output.template at<T>(z_idx, y_idx, x_idx, orderIdx ^ 2) =
76 Clamp<T, float>(RoundImplementationsFromYUV<float>(B), 0,
77 255);
78 }
79 }
