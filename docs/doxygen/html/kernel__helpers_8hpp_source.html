42static __host__ __device__ void blend_single_color(uchar4 &color, const uchar4 &color_in) {
43 int foreground_alpha = color_in.w;
44 int background_alpha = color.w;
45 int blend_alpha = ((background_alpha * (255 - foreground_alpha)) >> 8) + foreground_alpha;
46 color.x =
47 u8cast((((color.x * background_alpha * (255 - foreground_alpha)) >> 8) + (color_in.x * foreground_alpha)) /
48 blend_alpha);

50 u8cast((((color.y * background_alpha * (255 - foreground_alpha)) >> 8) + (color_in.y * foreground_alpha)) /
51 blend_alpha);
52 color.z =

54 blend_alpha);
55 color.w = blend_alpha;

149 const ConversionFactors factors) {
150 y = r * factors.a + g * factors.b + b * factors.c;

157 const ConversionFactors factors, float delta) {
