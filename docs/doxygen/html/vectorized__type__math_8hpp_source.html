74#define DEFINE_VECTOR_BINARY_FUNC(FUNC, NAME) \
75 template <typename T, typename U> \
76 __host__ __device__ auto NAME(T a, U b) { \
77 using RT = MakeType<decltype(FUNC(std::declval<BaseType<T>>(), std::declval<BaseType<U>>())), \
78 NumComponents<T> == 0 ? NumComponents<U> : NumComponents<T>>; \
79 if constexpr (NumComponents<T> == 0) { \
80 if constexpr (NumComponents<RT> == 1) \
81 return RT{FUNC(a, b.x)}; \
82 else if constexpr (NumComponents<RT> == 2) \
83 return RT{FUNC(a, b.x), FUNC(a, b.y)}; \
84 else if constexpr (NumComponents<RT> == 3) \
85 return RT{FUNC(a, b.x), FUNC(a, b.y), FUNC(a, b.z)}; \
86 else if constexpr (NumComponents<RT> == 4) \
87 return RT{FUNC(a, b.x), FUNC(a, b.y), FUNC(a, b.z), FUNC(a, b.w)}; \
88 \
89 } else if constexpr (NumComponents<U> == 0) { \
90 if constexpr (NumComponents<RT> == 1) \
91 return RT{FUNC(a.x, b)}; \
92 else if constexpr (NumComponents<RT> == 2) \
93 return RT{FUNC(a.x, b), FUNC(a.y, b)}; \
94 else if constexpr (NumComponents<RT> == 3) \
95 return RT{FUNC(a.x, b), FUNC(a.y, b), FUNC(a.z, b)}; \
96 else if constexpr (NumComponents<RT> == 4) \
97 return RT{FUNC(a.x, b), FUNC(a.y, b), FUNC(a.z, b), FUNC(a.w, b)}; \
98 \
99 } else { \
100 if constexpr (NumComponents<RT> == 1) \
101 return RT{FUNC(a.x, b.x)}; \
102 else if constexpr (NumComponents<RT> == 2) \
103 return RT{FUNC(a.x, b.x), FUNC(a.y, b.y)}; \
104 else if constexpr (NumComponents<RT> == 3) \

106 else if constexpr (NumComponents<RT> == 4) \

111#define DEFINE_VECTOR_UNARY_FUNC(FUNC, NAME) \
112 template <typename U, class = std::enable_if_t<HasTypeTraits<U>>> \
113 inline __host__ __device__ U NAME(U v) { \
114 if constexpr (NumComponents<U> == 0) \
115 return FUNC(v); \
116 else if constexpr (NumComponents<U> == 1) { \
117 return U{FUNC(v.x)}; \
118 } else if constexpr (NumComponents<U> == 2) { \
119 return U{FUNC(v.x), FUNC(v.y)}; \
120 } else if constexpr (NumComponents<U> == 3) { \
121 return U{FUNC(v.x), FUNC(v.y), FUNC(v.z)}; \
122 } else if constexpr (NumComponents<U> == 4) { \
123 return U{FUNC(v.x), FUNC(v.y), FUNC(v.z), FUNC(v.w)}; \
124 } \
