31template <typename T, typename SRC, typename DST>
32void binary_generic_kernel(SRC input, DST output, int64_t height, int64_t width,
33 int64_t channels, const uint8_t *thresh,
34 const uint8_t *maxVal, const int64_t maxBatchSize) {
35 for (int z_idx = 0; z_idx < maxBatchSize; z_idx++) {
36 for (int y_idx = 0; y_idx < height; y_idx++) {
37 for (int x_idx = 0; x_idx < width; x_idx++) {
38 uint8_t th = thresh[z_idx];
39 uint8_t mv = maxVal[z_idx];
40 for (int i = 0; i < channels; i++) {
41 auto ip = input.template at<T>(z_idx, y_idx, x_idx, i);
42 output.template at<T>(z_idx, y_idx, x_idx, i) =
43 ip > th ? mv : 0;
44 }
45 }
46 }
