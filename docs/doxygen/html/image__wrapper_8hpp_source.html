36template <typename T>
37class ImageWrapper {
38 public:
39 using ValueType = T;
40
46 ImageWrapper(const Tensor& tensor) {
47 if (tensor.layout() != eTensorLayout::TENSOR_LAYOUT_NHWC &&
48 tensor.layout() != eTensorLayout::TENSOR_LAYOUT_NCHW &&
49 tensor.layout() != eTensorLayout::TENSOR_LAYOUT_HWC) {
50 throw Exception("The given tensor layout is not supported for ImageWrapper", eStatusType::NOT_IMPLEMENTED);
51 }
52
53 // Copy tensor data into image tensor descriptor
54 TensorDataStrided tdata = tensor.exportData<TensorDataStrided>();
55 ImageShape indexes = {tensor.layout().batch_index(), tensor.layout().height_index(),
56 tensor.layout().width_index(), tensor.layout().channels_index()};
57
58 // Handle HWC layout, which doesn't have shapes/strides for the batch dimension. We set the batch shape to 1 and
59 // the strides to 0.
60 int64_t num_batches = indexes.n < 0 ? 1 : tensor.shape(indexes.n);
61 int64_t batch_stride = indexes.n < 0 ? 0 : tdata.stride(indexes.n);
62
63 shape = {num_batches, tdata.shape(indexes.h), tdata.shape(indexes.w), tdata.shape(indexes.c)};
64 stride = {batch_stride, tdata.stride(indexes.h), tdata.stride(indexes.w), tdata.stride(indexes.c)};
65 data = static_cast<unsigned char*>(tdata.basePtr());
66 }
67
77 __device__ __host__ T& at(int64_t n, int64_t h, int64_t w, int64_t c) {
78 return *(reinterpret_cast<T*>(data + (stride.n * n) + (stride.h * h) + (stride.w * w) + (stride.c * c)));
79 }
80
81 __device__ __host__ const T at(int64_t n, int64_t h, int64_t w, int64_t c) const {
82 return *(reinterpret_cast<T*>(data + (stride.n * n) + (stride.h * h) + (stride.w * w) + (stride.c * c)));
83 }
84
90 __device__ __host__ inline int64_t height() const { return shape.h; }
91
97 __device__ __host__ inline int64_t width() const { return shape.w; }
98
104 __device__ __host__ inline int64_t batches() const { return shape.n; }
105
111 __device__ __host__ inline int64_t channels() const { return shape.c; }
112
113 private:
114 struct ImageShape {
115 int64_t n, h, w, c;
116 };
117
118 ImageShape shape;
119 ImageShape stride;
120
121 unsigned char* data;
122};
