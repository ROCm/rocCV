31template <typename T, typename SRC, typename DST>
32__global__ void rgb_or_bgr_to_yuv(SRC input, DST output, int64_t width,
33 int64_t height, int64_t batch_size,
34 int orderIdx, float delta) {
35 const auto x_idx = threadIdx.x + blockIdx.x * blockDim.x;
36 const auto y_idx = threadIdx.y + blockIdx.y * blockDim.y;
37 const auto z_idx = threadIdx.z + blockIdx.z * blockDim.z;
38 if (x_idx < width && y_idx < height && z_idx < batch_size) {
39 T R = input.template at<T>(z_idx, y_idx, x_idx, orderIdx);
40 T G = input.template at<T>(z_idx, y_idx, x_idx, 1);
41 T B = input.template at<T>(z_idx, y_idx, x_idx, orderIdx ^ 2);
42
43 float Y = R * 0.299f + G * 0.587f + B * 0.114f;
44 float Cr = (R - Y) * 0.877f + delta;
45 float Cb = (B - Y) * 0.492f + delta;
46
47 output.template at<T>(z_idx, y_idx, x_idx, 0) =
48 RoundImplementationsToYUV<float>(Y);
49 output.template at<T>(z_idx, y_idx, x_idx, 1) =
50 RoundImplementationsToYUV<float>(Cb);
51 output.template at<T>(z_idx, y_idx, x_idx, 2) =
52 RoundImplementationsToYUV<float>(Cr);
53 }

57__global__ void yuv_to_rgb_or_bgr(SRC input, DST output, int64_t width,
58 int64_t height, int64_t batch_size,
59 int orderIdx, float delta) {
60 const int x_idx = threadIdx.x + blockDim.x * blockIdx.x;
61 const int y_idx = threadIdx.y + blockDim.y * blockIdx.y;
62 const int z_idx = threadIdx.z + blockDim.z * blockIdx.z;
63
64 if (x_idx < width && y_idx < height && z_idx < batch_size) {
65 T Y = input.template at<T>(z_idx, y_idx, x_idx, 0);
66 T Cb = input.template at<T>(z_idx, y_idx, x_idx, 1);
67 T Cr = input.template at<T>(z_idx, y_idx, x_idx, 2);
68
69 float B = Y + (Cb - delta) * 2.032f;
70 float G = Y + (Cb - delta) * -0.395f + (Cr - delta) * -0.581f;
71 float R = Y + (Cr - delta) * 1.140f;
72
73 output.template at<T>(z_idx, y_idx, x_idx, orderIdx) =
74 Clamp<T, float>(RoundImplementationsFromYUV<float>(R), 0, 255);
75 output.template at<T>(z_idx, y_idx, x_idx, 1) =
76 Clamp<T, float>(RoundImplementationsFromYUV<float>(G), 0, 255);
77 output.template at<T>(z_idx, y_idx, x_idx, orderIdx ^ 2) =
78 Clamp<T, float>(RoundImplementationsFromYUV<float>(B), 0, 255);
79 }
