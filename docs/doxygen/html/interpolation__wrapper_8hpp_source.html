61 inline __device__ __host__ const T at(int64_t n, float h, float w, int64_t c) const {
62 if constexpr (I == eInterpolationType::INTERP_TYPE_NEAREST) {
63 // Nearest neighbor interpolation implementation
64 return m_desc.at(n, static_cast<int64_t>(std::round(h)), static_cast<int64_t>(std::round(w)), c);
65 } else if constexpr (I == eInterpolationType::INTERP_TYPE_LINEAR) {
66 // Bilinear interpolation implementation
67 // v1 -- v2
68 // - -
69 // v3 -- v4
70
71 // Intermediate values should be float vectors to avoid overflow
72 using WorkType = detail::MakeType<float, detail::NumElements<T>>;
73
74 int64_t x0 = static_cast<int64_t>(floor(w));
75 int64_t x1 = x0 + 1;
76 int64_t y0 = static_cast<int64_t>(floor(h));
77 int64_t y1 = y0 + 1;
78
79 auto v1 = detail::StaticCast<WorkType>(m_desc.at(n, y0, x0, c));
80 auto v2 = detail::StaticCast<WorkType>(m_desc.at(n, y0, x1, c));
81 auto v3 = detail::StaticCast<WorkType>(m_desc.at(n, y1, x0, c));
82 auto v4 = detail::StaticCast<WorkType>(m_desc.at(n, y1, x1, c));
83
84 auto q1 = v1 * (x1 - w) + v2 * (w - x0);
85 auto q2 = v3 * (x1 - w) + v4 * (w - x0);
86 auto q = q1 * (y1 - h) + q2 * (h - y0);
87
88 return detail::SaturateCast<T>(q);
89 }
90
91 // TODO: Support other interpolation methods.
92
93 else {
94 static_assert(false, "Provided interpolation type is not supported.");
95 }
96 }
