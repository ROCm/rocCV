201// Matrix inversion
202template <class T>
203constexpr __host__ __device__ Matrix<T, 3, 3> inv(const Matrix<T, 3, 3> &m, const T &d) {
204 Matrix<T, 3, 3> A;
205 A[0][0] = (m[1][1] * m[2][2] - m[1][2] * m[2][1]) / d;
206 A[0][1] = -(m[0][1] * m[2][2] - m[0][2] * m[2][1]) / d;
207 A[0][2] = (m[0][1] * m[1][2] - m[0][2] * m[1][1]) / d;
208 A[1][0] = -(m[1][0] * m[2][2] - m[1][2] * m[2][0]) / d;
209 A[1][1] = (m[0][0] * m[2][2] - m[0][2] * m[2][0]) / d;
210 A[1][2] = -(m[0][0] * m[1][2] - m[0][2] * m[1][0]) / d;
211 A[2][0] = (m[1][0] * m[2][1] - m[1][1] * m[2][0]) / d;
212 A[2][1] = -(m[0][0] * m[2][1] - m[0][1] * m[2][0]) / d;
213 A[2][2] = (m[0][0] * m[1][1] - m[0][1] * m[1][0]) / d;
214
215 return A;
216}

253template <class F = float, class T, int N>
254__host__ __device__ bool lu_inplace(Matrix<T, N, N> &m, Vector<int, N> &p) {
255 Vector<F, N> v;
256
257#pragma unroll
258 for (int i = 0; i < N; ++i) {
259 F big = 0;
260
261#pragma unroll
262 for (int j = 0; j < N; ++j) {
263 big = max(big, abs(m[i][j]));
264 }
265
266 if (big == 0) {
267 return false;
268 }
269
270 v[i] = 1.0 / big;
271 }
272
273#pragma unroll
274 for (int k = 0; k < N; ++k) {
275 F big = 0;
276 int imax = k;
277
278#pragma unroll
279 for (int i = k; i < N; ++i) {
280 F aux = v[i] * abs(m[i][k]);
281
282 if (aux > big) {
283 big = aux;
284 imax = i;
285 }
286 }
287
288 if (k != imax) {
289 swap(m[imax], m[k]);
290
291 v[imax] = v[k];
292 }
293
294 p[k] = imax;
295
296 if (m[k][k] == 0) {
297 return false;
298 }
299
300#pragma unroll
301 for (int i = k + 1; i < N; ++i) {
302 T aux = m[i][k] /= m[k][k];
303
304#pragma unroll
305 for (int j = k + 1; j < N; ++j) {
306 m[i][j] -= aux * m[k][j];
307 }
308 }
309 }
310
311 return true;
312}

314template <class T, int N>
315__host__ __device__ void solve_inplace(const Matrix<T, N, N> &lu, const Vector<int, N> &p, Vector<T, N> &b) {
316 int ii = -1;
317
318#pragma unroll
319 for (int i = 0; i < N; ++i) {
320 int ip = p[i];
321 T sum = b[ip];
322 b[ip] = b[i];
323
324 if (ii >= 0) {
325 for (int j = ii; j < i; ++j) {
326 sum -= lu[i][j] * b[j];
327 }
328 } else if (sum != 0) {
329 ii = i;
330 }
331
332 b[i] = sum;
333 }
334
335#pragma unroll
336 for (int i = N - 1; i >= 0; --i) {
337 T sum = b[i];
338
339#pragma unroll
340 for (int j = i + 1; j < N; ++j) {
341 sum -= lu[i][j] * b[j];
342 }
343
344 b[i] = sum / lu[i][i];
345 }
346}
