29inline __device__ __host__ float ComputeIoU(const short4 &boxA, const short4 &boxB) {
30 int aInterLeft = std::max(boxA.x, boxB.x);
31 int bInterTop = std::max(boxA.y, boxB.y);
32 int aInterRight = std::min(boxA.x + boxA.z, boxB.z + boxB.z);
33 int bInterBottom = std::min(boxA.y + boxA.w, boxB.y + boxB.w);
34 int widthInter = aInterRight - aInterLeft;
35 int heightInter = bInterBottom - bInterTop;
36 float interArea = widthInter * heightInter;
37 float iou = 0.0f;
38
39 if (widthInter > 0.0f && heightInter > 0.0f) {
40 float unionArea = ComputeArea(boxA) + ComputeArea(boxB) - interArea;
41 if (unionArea > 0.0f) {
42 iou = interArea / unionArea;
43 }
44 }
45
46 return iou;
47}
